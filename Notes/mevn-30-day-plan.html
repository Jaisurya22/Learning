Step 2: Create Your First File

In VS Code's file explorer (left sidebar), right-click empty space → New File → name it day1.js

Step 3: Write This Code

Type exactly this into day1.js:

// ============================================
// DAY 1: Variables, Types, Functions
// In Java you wrote:  int age = 25;
// In JavaScript:      let age = 25;
// ============================================

// --- VARIABLES ---
let name = "Alex";        // String (like Java String)
const age = 25;            // Number (like Java int, but also works for decimals)
let isLearning = true;   // Boolean (like Java boolean)

// Print to console — Java uses System.out.println()
console.log(name);
console.log("Age: " + age);

// --- TYPES (JavaScript is flexible) ---
let price = 19.99;         // number — no int vs double in JS
let nothing = null;        // null — empty on purpose
let notDefined;              // undefined — declared but no value

console.log(typeof price);      // prints: "number"
console.log(typeof notDefined); // prints: "undefined"

// --- FUNCTIONS (Java: public void sayHello(String name)) ---
function sayHello(name) {
    return "Hello " + name + "!";
}

let message = sayHello("Alex");
console.log(message);

// --- ARROW FUNCTION (shorter way — JS-only) ---
const sayHi = (name) => {
    return "Hi " + name + "!";
};
console.log(sayHi("Alex"));

// One-line arrow (even shorter — no curly braces needed)
const double = (x) => x * 2;
console.log(double(5)); // prints: 10

Step 4: Run It

You should see your name, age, and function outputs printed.

If you see "node is not recognized", you didn't install Node.js correctly. Restart VS Code after installing Node.js, or restart your computer.

// ============================================

Day 4 Express Routes + JSON APIs

Create a REST API that returns JSON data — the backbone of every backend.
const express = require("express");
const app = express();

app.use(express.json()); // tells Express to understand JSON body data

// In-memory "database" (we'll replace with MongoDB later)
let bugs = [
    { id: 1, title: "Login broken", status: "open" },
    { id: 2, title: "UI misaligned", status: "resolved" }
];
let nextId = 3;

// GET /api/bugs — list all bugs
app.get("/api/bugs", (req, res) => {
    res.json(bugs); // sends JSON (browser shows it as data)
});

// GET /api/bugs/:id — get one bug by ID
app.get("/api/bugs/:id", (req, res) => {
    const bug = bugs.find((b) => b.id === Number(req.params.id));
    if (!bug) {
        return res.status(404).json({ error: "Bug not found" });
    }
    res.json(bug);
});

// POST /api/bugs — create a new bug
app.post("/api/bugs", (req, res) => {
    const bug = {
        id: nextId++,
        title: req.body.title,     // from JSON sent by client
        status: req.body.status || "open"
    };
    bugs.push(bug);
    res.status(201).json(bug); // 201 = Created
});

// PUT /api/bugs/:id — update a bug
app.put("/api/bugs/:id", (req, res) => {
    const bug = bugs.find((b) => b.id === Number(req.params.id));
    if (!bug) {
        return res.status(404).json({ error: "Bug not found" });
    }
    bug.title = req.body.title || bug.title;
    bug.status = req.body.status || bug.status;
    res.json(bug);
});

// DELETE /api/bugs/:id — delete a bug
app.delete("/api/bugs/:id", (req, res) => {
    bugs = bugs.filter((b) => b.id !== Number(req.params.id));
    res.status(204).send(); // 204 = No content (successfully deleted)
});

app.listen(5000, () => console.log("API running on :5000"));

const bugSchema = new mongoose.Schema({

Create server.js with MongoDB:

const express = require("express");
const mongoose = require("mongoose");
const Bug = require("./models/Bug");
const app = express();

app.use(express.json());

// Connect to MongoDB
// Replace with YOUR connection string from Atlas (or use local)
mongoose.connect("mongodb://127.0.0.1:27017/bugtracker")
    .then(() => console.log("MongoDB connected"))
    .catch(err => console.log(err));

// CREATE a bug (using Mongoose)
app.post("/api/bugs", async (req, res) => {
    try {
        const bug = await Bug.create(req.body); // creates AND saves to DB
        res.status(201).json(bug);
    } catch (err) {
        res.status(400).json({ error: err.message });
    }
});

// READ all bugs
app.get("/api/bugs", async (req, res) => {
    const bugs = await Bug.find(); // SELECT * FROM bugs
    res.json(bugs);
});

// READ one bug
app.get("/api/bugs/:id", async (req, res) => {
    const bug = await Bug.findById(req.params.id);
    if (!bug) return res.status(404).json({ error: "Not found" });
    res.json(bug);
});

// UPDATE a bug
app.put("/api/bugs/:id", async (req, res) => {
    const bug = await Bug.findByIdAndUpdate(req.params.id, req.body, { new: true });
    if (!bug) return res.status(404).json({ error: "Not found" });
    res.json(bug);
});

// DELETE a bug
app.delete("/api/bugs/:id", async (req, res) => {
    const bug = await Bug.findByIdAndDelete(req.params.id);
    if (!bug) return res.status(404).json({ error: "Not found" });
    res.status(204).send();
});

app.listen(5000, () => console.log("API with DB running"));
async/await explained: await says "wait for this database operation to finish before moving on". async marks the function as using await. In Java, you'd use CompletableFuture — same idea.

Day 7 Detail Auth: Register & Login (JWT)

This is the hardest part of any backend. Take it slow.

Step 1: Install packages

npm install bcryptjs jsonwebtoken

Step 2: User model

const mongoose = require("mongoose");
const bcrypt = require("bcryptjs");

const userSchema = new mongoose.Schema({
    name: { type: String, required: [true, "Name is required"] },
    email: { type: String, required: true, unique: true, lowercase: true },
    password: { type: String, required: true, minlength: 6, select: false },
    role: { type: String, enum: ["admin", "member"], default: "member" }
}, { timestamps: true }); // automatically adds createdAt, updatedAt

// Hash password BEFORE saving (like a Java @PrePersist)
userSchema.pre("save", async function (next) {
    if (!this.isModified("password")) return next();
    this.password = await bcrypt.hash(this.password, 12); // 12 = salt rounds
    next();
});

// Method to compare passwords (like Java .equals)
userSchema.methods.comparePassword = async function (candidatePassword) {
    return await bcrypt.compare(candidatePassword, this.password);
};

module.exports = mongoose.model("User", userSchema);

Step 3: Auth controller

const User = require("../models/User");
const jwt = require("jsonwebtoken");

// Helper: generate JWT token
const signToken = (userId) => {
    return jwt.sign({ id: userId }, process.env.JWT_SECRET, { expiresIn: "15m" });
};

exports.register = async (req, res) => {
    const { name, email, password } = req.body;
    const user = await User.create({ name, email, password });
    const token = signToken(user._id);
    res.status(201).json({ token, user: { id: user._id, name, email, role: user.role } });
};

exports.login = async (req, res) => {
    const { email, password } = req.body;
    const user = await User.findOne({ email }).select("+password"); // include password (excluded by default)
    if (!user || !(await user.comparePassword(password))) {
        return res.status(401).json({ error: "Invalid email or password" });
    }
    const token = signToken(user._id);
    res.json({ token, user: { id: user._id, name: user.name, email, role: user.role } });
};

Step 4: Auth middleware

const jwt = require("jsonwebtoken");
const User = require("../models/User");

module.exports = async (req, res, next) => {
    try {
        const token = req.headers.authorization?.replace("Bearer ", "");
        if (!token) throw new Error("No token provided");

        const decoded = jwt.verify(token, process.env.JWT_SECRET);
        req.user = await User.findById(decoded.id);
        if (!req.user) throw new Error("User not found");
        next();
    } catch (err) {
        res.status(401).json({ error: "Authentication required" });
    }
};
How JWT works: User logs in → server creates a signed token (like a digital passport) → client stores it → sends it with every request in the Authorization: Bearer <token> header → server verifies it → if valid, processes request.

Day 25 Detail WebSockets with Socket.io

Backend: Add Socket.io

In server.js, add:

const http = require("http");
const { Server } = require("socket.io");

const server = http.createServer(app); // wrap Express in HTTP server
const io = new Server(server, { cors: { origin: "*" } });

// When a client connects
io.on("connection", (socket) => {
    console.log("Client connected:", socket.id);

    socket.on("join-project", (projectId) => {
        socket.join(`project:${projectId}`); // join a room
    });
});

// Emit events when bugs change (in your controller)
// io.to(`project:${projectId}`).emit("bug:created", newBug);
// io.to(`project:${projectId}`).emit("bug:updated", updatedBug);

// Use server.listen instead of app.listen
server.listen(PORT, () => console.log("Server with WebSocket running"));

Frontend: Connect

import { io } from "socket.io-client"

const socket = io("http://localhost:5000")

// Join project room when viewing a project
socket.emit("join-project", projectId)

// Listen for real-time updates
socket.on("bug:created", (bug) => {
    bugs.value.unshift(bug) // add to top of list
})

socket.on("bug:updated", (bug) => {
    const idx = bugs.value.findIndex(b => b._id === bug._id)
    if (idx !== -1) bugs.value[idx] = bug
})
What does this do? Without WebSocket, you have to refresh the page to see new bugs. With WebSocket, when anyone creates a bug, it appears instantly on everyone's screen — like Google Docs real-time editing.

Troubleshooting (When You Get Stuck)

ErrorMost Likely CauseFix
Cannot find module 'express'You forgot to npm install expressRun npm install express in the folder where package.json is
ECONNREFUSED ::1:27017MongoDB isn't runningStart MongoDB: mongod or check Services panel. Or use Atlas
401 UnauthorizedMissing/invalid tokenCheck Authorization: Bearer <token> header is set. Token might be expired
404 Not FoundWrong URLCheck your route path. /api/bugs vs /api/bugs/ matters
ValidationErrorMissing required fieldCheck your schema. title is required? Send it in the request body
CORS error (browser)Frontend port different from backendAdd npm install cors then app.use(cors()) in server.js
npm run dev not foundYou're in the wrong foldercd frontend first. Vue commands run from the frontend/ folder
General debugging: Add console.log() everywhere. Check what data is coming in/out. The terminal shows errors — read them carefully, they tell you exactly what's wrong and where.

How To Use This Page

  1. Do one day per day. Don't skip ahead — each day builds on the previous
  2. Type the code yourself. Don't copy-paste. Typing builds muscle memory
  3. If you're stuck for >30min: Skip to the next day. Or re-read the current day's code from scratch
  4. Test after every step. After adding a route, test it. After adding a component, view it
  5. Break things on purpose. Change a variable name, see what error you get, then fix it. This is how you learn
  6. Use AI for explanations. If a concept doesn't click, ask a coding AI: "Explain [concept] like I'm 12"
  7. Save your work. After each day, commit to git: git add . && git commit -m "Day X"
